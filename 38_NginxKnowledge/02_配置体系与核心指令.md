## 一、配置文件层级结构

```nginx
# ① 全局块 ─ 影响 Nginx 整体
worker_processes  auto;
worker_rlimit_nofile 65535;
error_log  /var/log/nginx/error.log warn;
pid        /var/run/nginx.pid;

# ② events 块 ─ 网络连接配置
events {
    worker_connections  10240;
    use epoll;
    multi_accept on;
}

# ③ http 块 ─ 所有 HTTP/HTTPS 服务配置
http {
    include       mime.types;
    default_type  application/octet-stream;
    sendfile      on;
    keepalive_timeout 75s;
    
    # ④ upstream 块 ─ 后端服务器组
    upstream backend {
        server 127.0.0.1:8080 weight=5;
        server 127.0.0.1:8081 weight=3;
        keepalive 32;
    }
    
    # ⑤ server 块 ─ 虚拟主机
    server {
        listen       80;
        server_name  example.com www.example.com;
        
        # ⑥ location 块 ─ URI 路由规则
        location / {
            proxy_pass http://backend;
        }
    }
}
```

**就近覆盖原则**：内层块配置可覆盖外层同名配置。  
**拆分管理**：`include /etc/nginx/conf.d/*.conf;` — 将 server 块拆入单独文件。

## 二、虚拟主机（Server）

### 2.1 三种虚拟主机类型

```nginx
# ① 基于域名（最常用）
server {
    listen 80;
    server_name www.site-a.com site-a.com;
    root /var/www/site-a;
}

# ② 基于端口
server {
    listen 8080;
    server_name localhost;
    root /var/www/app1;
}

# ③ 基于 IP
server {
    listen 192.168.1.10:80;
    server_name _;
    root /var/www/site1;
}

# 默认 server（兜底，防止未知域名请求被处理）
server {
    listen 80 default_server;
    server_name _;
    return 444;  # 直接关闭连接
}
```

### 2.2 server_name 匹配优先级

1. **精确名称**：`server_name www.example.com;`
2. **`*` 开头通配符**：`server_name *.example.com;`
3. **`*` 结尾通配符**：`server_name www.example.*;`
4. **正则表达式**：`server_name ~^www\d+\.example\.com$;`（按配置顺序）
5. **default_server**：所有规则不匹配时的兜底

## 三、location 指令详解

### 3.1 匹配修饰符与优先级

| 优先级 | 语法 | 说明 |
|--------|------|------|
| 1（最高） | `location = /uri` | 精确匹配，命中立即停止检查 |
| 2 | `location ^~ /prefix/` | 最长前缀命中后，**阻止正则检查** |
| 3 | `location ~ /regex` | 正则（区分大小写），按配置顺序第一个命中 |
| 3 | `location ~* /regex` | 正则（不区分大小写） |
| 4（最低） | `location /prefix/` | 普通前缀，选最长命中，但仍会检查正则 |

### 3.2 完整匹配流程

```
请求 URI
   │
   ├─① 检查 = 精确匹配 ──── 命中 ──→ 直接使用
   │
   ├─② 扫描所有前缀，记录最长命中的 location
   │
   ├─③ 最长命中是 ^~ ──── 是 ──→ 直接使用，不检查正则
   │
   ├─④ 按配置文件顺序检查正则 ──── 第一个命中 ──→ 使用该正则 location
   │
   └─⑤ 无正则命中 ──→ 使用步骤②的最长前缀 location
```

### 3.3 实战示例

```nginx
server {
    location = /api/health    { return 200 "ok"; }         # 精确，健康检查
    location ^~ /static/      { root /var/www; }           # 前缀阻断，静态资源
    location ~* \.(jpg|png|gif|ico|css|js)$ { expires 30d; }  # 正则，资源缓存
    location /api/            { proxy_pass http://backend; } # 普通前缀，API代理
    location /                { try_files $uri $uri/ =404; } # 兜底
}
```

## 四、核心性能指令

### 4.1 超时配置

```nginx
http {
    # 客户端超时
    client_header_timeout  15s;    # 读取请求头超时（默认 60s）
    client_body_timeout    60s;    # 读取请求体超时
    send_timeout           60s;    # 发送响应超时（两次写操作间隔）
    keepalive_timeout      75s;    # Keep-Alive 连接保持时间
    
    # 代理超时（最常调）
    proxy_connect_timeout  5s;     # 连接上游超时（不宜超过 10s）
    proxy_send_timeout     60s;    # 向上游发送请求超时
    proxy_read_timeout     60s;    # 从上游读取响应超时
    
    upstream backend {
        server 127.0.0.1:8080;
        keepalive 32;              # 与后端保持 32 个长连接
        keepalive_timeout 60s;     # 长连接空闲超时
    }
}
```

### 4.2 缓冲区配置

```nginx
# 代理缓冲（防止后端响应慢导致连接长时间占用）
proxy_buffer_size          4k;    # 响应头缓冲
proxy_buffers              8 8k;  # 响应体缓冲（8个8k缓冲）
proxy_busy_buffers_size    16k;   # 同时发送给客户端的缓冲大小
proxy_temp_file_write_size 64k;   # 写临时文件时的大小

# 客户端请求缓冲
client_body_buffer_size    16k;   # 请求体缓冲（超出写磁盘）
client_max_body_size       100m;  # 允许上传的最大请求体（0=不限）
```

## 五、HTTPS 完整配置

```nginx
server {
    listen 443 ssl http2;
    server_name www.example.com;
    
    # 证书（包含完整证书链）
    ssl_certificate     /etc/nginx/ssl/fullchain.pem;
    ssl_certificate_key /etc/nginx/ssl/privkey.pem;
    
    # 只允许 TLS 1.2 + TLS 1.3（禁用 SSLv3/TLS1.0/TLS1.1）
    ssl_protocols TLSv1.2 TLSv1.3;
    
    # Mozilla 推荐的加密套件
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384:ECDHE-ECDSA-CHACHA20-POLY1305:ECDHE-RSA-CHACHA20-POLY1305;
    ssl_prefer_server_ciphers off;  # TLS 1.3 下 off 更安全
    
    # Session 缓存（复用 TLS 握手，减少延迟）
    ssl_session_cache   shared:SSL:10m;  # 10MB，约 4 万个会话
    ssl_session_timeout 1d;
    ssl_session_tickets off;             # 关闭 tickets（提升前向安全）
    
    # OCSP Stapling（证书在线状态，加速 TLS 握手）
    ssl_stapling on;
    ssl_stapling_verify on;
    ssl_trusted_certificate /etc/nginx/ssl/chain.pem;
    resolver 8.8.8.8 valid=300s;
    
    # 安全响应头
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains; preload" always;
    add_header X-Content-Type-Options nosniff always;
    add_header X-Frame-Options DENY always;
    add_header X-XSS-Protection "1; mode=block" always;
    
    location / {
        proxy_pass http://backend;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}

# HTTP → HTTPS 跳转
server {
    listen 80;
    server_name www.example.com;
    return 301 https://$host$request_uri;
}
```

## 六、访问控制指令

```nginx
# IP 白名单（从上到下，第一条命中生效）
location /admin {
    allow 10.0.0.0/8;
    allow 192.168.1.100;
    allow 127.0.0.1;
    deny  all;
}

# 状态监控页面
location /nginx_status {
    stub_status;
    allow 127.0.0.1;
    deny  all;
}

# 防盗链（referer 验证）
location ~* \.(jpg|png|gif)$ {
    valid_referers none blocked server_names *.mysite.com;
    if ($invalid_referer) {
        return 403;  # 或返回水印图片
    }
}
```

## 七、日志配置

```nginx
http {
    # 标准日志格式
    log_format main '$remote_addr - $remote_user [$time_local] "$request" '
                    '$status $body_bytes_sent "$http_referer" '
                    '"$http_user_agent" $request_time $upstream_response_time';
    
    # JSON 格式（推荐，方便 ELK 收集）
    log_format json_log escape=json
        '{"@timestamp":"$time_iso8601",'
        '"remote_addr":"$remote_addr",'
        '"method":"$request_method",'
        '"uri":"$request_uri",'
        '"status":$status,'
        '"request_time":$request_time,'
        '"upstream_rt":"$upstream_response_time",'
        '"bytes":$body_bytes_sent,'
        '"referer":"$http_referer",'
        '"ua":"$http_user_agent"}';
    
    access_log /var/log/nginx/access.log json_log buffer=32k flush=5s;
    error_log  /var/log/nginx/error.log  warn;
    
    # 关闭 access_log（高性能场景，减少 I/O）
    # access_log off;
}

server {
    # 单独 server 的日志（覆盖全局）
    access_log /var/log/nginx/api.access.log main;
}
```

**常用变量**：
- `$request_time`：请求总耗时（含上游等待）
- `$upstream_response_time`：上游响应时间
- `$upstream_addr`：实际转发到的上游地址（多个用逗号分隔）
- `$http_x_forwarded_for`：原始客户端 IP 链
