## 一、静态资源性能优化

### 1.1 sendfile 零拷贝

```nginx
http {
    sendfile       on;   # 启用零拷贝（内核直接 page cache → socket，不走用户空间）
    tcp_nopush     on;   # 配合 sendfile：将响应头和文件数据合并成一个 TCP 包发送
    tcp_nodelay    on;   # 禁用 Nagle 算法：小包立即发送（适合实时性高的场景如 WebSocket）
}
```

**sendfile 工作原理**：
```
传统 I/O:  磁盘 → 内核 page cache → 用户空间 → 内核 socket buffer → 网卡
sendfile:  磁盘 → 内核 page cache ──────────────→ socket buffer → 网卡（减少 2 次拷贝和 2 次上下文切换）
```

**注意**：
- `tcp_nopush` 和 `tcp_nodelay` 同时开启时，Nginx 先用 `nopush` 积攒数据，最后一包用 `nodelay` 立即发送
- 动态内容（proxy_pass）`sendfile` 无效（没有磁盘文件可以直接发送）

### 1.2 Gzip 压缩

```nginx
http {
    gzip on;
    gzip_vary on;              # 添加 Vary: Accept-Encoding 头，支持CDN区分缓存
    gzip_proxied any;          # 对代理请求也压缩
    gzip_comp_level 6;         # 压缩级别（1-9，6 是性能与压缩率的最佳平衡）
    gzip_min_length 1024;      # 小于 1KB 的响应不压缩（压缩开销 > 收益）
    gzip_buffers 16 8k;        # 压缩缓冲区
    
    # 需要压缩的 MIME 类型（图片/视频/音频不压缩，已是压缩格式）
    gzip_types
        text/plain text/css text/xml text/javascript
        application/json application/javascript application/xml
        application/x-font-ttf font/opentype image/svg+xml;
    
    # 不压缩古老浏览器（MSIE6 压缩会乱码）
    gzip_disable "msie6";
}
```

**压缩率参考**：
- HTML/CSS/JS：压缩 60~80%
- JSON API 响应：压缩 50~70%
- 图片（JPG/PNG）：几乎无效，反而增加 CPU 开销

### 1.3 浏览器缓存

```nginx
# 静态资源强缓存（hash 版本文件名，内容变则文件名变）
location ~* \.(css|js)$ {
    expires 1y;
    add_header Cache-Control "public, immutable";  # immutable: 即使有 max-age 也不验证
}

# 图片缓存
location ~* \.(jpg|jpeg|png|gif|ico|webp|svg)$ {
    expires 30d;
    add_header Cache-Control "public";
}

# 字体文件
location ~* \.(woff|woff2|ttf|eot)$ {
    expires 1y;
    add_header Cache-Control "public, immutable";
    add_header Access-Control-Allow-Origin "*";  # 字体需要 CORS（浏览器限制）
}

# HTML：不缓存（入口文件要及时更新）
location ~* \.html$ {
    expires -1;
    add_header Cache-Control "no-store, no-cache, must-revalidate";
}
```

## 二、限流与并发控制

### 2.1 limit_req（请求速率限制）

```nginx
http {
    # 定义限流区域（10MB 内存约可存 16 万个 IP 的状态）
    limit_req_zone $binary_remote_addr zone=api_limit:10m rate=100r/s;
    limit_req_zone $binary_remote_addr zone=login_limit:10m rate=5r/m;  # 登录接口严格限制
    limit_req_zone $http_x_forwarded_for zone=forward_limit:10m rate=50r/s;  # 基于真实IP
    
    server {
        # API 接口：100 QPS，允许突发 20 个请求排队
        location /api/ {
            limit_req zone=api_limit burst=20 nodelay;
            # nodelay: 突发请求立即处理（不等待），超出 burst 才返回 503
            # delay=5: 超出 rate 时最多 5 个请求可以延迟处理（其余立即拒绝）
            limit_req_status 429;  # 超限时返回 429 而非默认 503
            proxy_pass http://backend;
        }
        
        # 登录接口：严格限制（防暴力破解）
        location /auth/login {
            limit_req zone=login_limit burst=3 nodelay;
            limit_req_status 429;
            proxy_pass http://backend;
        }
    }
}
```

**limit_req 算法**：令牌桶（Token Bucket）
- `rate=100r/s`：每 10ms 放一个令牌（100/s = 1/10ms）
- `burst=20`：令牌桶容量为 20（允许短时间突发）
- `nodelay`：burst 内的请求立即处理，不排队等待；无 nodelay 则排队等待（延迟响应）

### 2.2 limit_conn（并发连接数限制）

```nginx
http {
    limit_conn_zone $binary_remote_addr zone=conn_limit:10m;
    limit_conn_zone $server_name        zone=server_limit:10m;
    
    server {
        limit_conn conn_limit  20;    # 每个 IP 最多 20 个并发连接
        limit_conn server_limit 1000; # 整个 server 最多 1000 个并发连接
        limit_conn_status 503;
        
        # 针对下载接口限速（防止单个用户占用太多带宽）
        location /download/ {
            limit_conn conn_limit 5;    # 每 IP 最多 5 个并发下载
            limit_rate 512k;            # 每个连接限速 512KB/s
            limit_rate_after 10m;       # 前 10MB 不限速（提升体验）
            proxy_pass http://backend;
        }
    }
}
```

## 三、安全防护

### 3.1 隐藏版本信息

```nginx
http {
    server_tokens off;  # 隐藏 Server: nginx/1.24.0 中的版本号 → Server: nginx
}
# 彻底修改 Server 头需要第三方模块 ngx_http_headers_more_module：
# more_set_headers "Server: WebServer";
```

### 3.2 DDoS/CC 防护

```nginx
http {
    # 限制每 IP 的请求速率
    limit_req_zone $binary_remote_addr zone=ddos_limit:20m rate=30r/s;
    
    # 限制每 IP 并发连接数
    limit_conn_zone $binary_remote_addr zone=ddos_conn:20m;
    
    # IP 黑名单（使用 geo 模块）
    geo $blocked_ip {
        default        0;
        1.2.3.4        1;  # 手动封禁的 IP
        10.0.0.0/8     0;  # 内网 IP 不限制
    }
    
    server {
        # 封禁黑名单 IP
        if ($blocked_ip) {
            return 403;
        }
        
        limit_conn  ddos_conn  50;       # 每 IP 最多 50 个并发
        limit_req   zone=ddos_limit burst=100 nodelay;
        
        # 客户端请求体大小限制（防止大请求耗尽内存）
        client_max_body_size 10m;
        client_body_timeout  15s;
        client_header_timeout 15s;
        
        # 慢速攻击防护（Slowloris）
        # 通过 client_header_timeout 和 client_body_timeout 限制
    }
}
```

### 3.3 SSL/TLS 安全策略

```nginx
# 完整的 TLS 安全配置（Mozilla Modern 级别）
ssl_protocols TLSv1.2 TLSv1.3;
ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384:ECDHE-ECDSA-CHACHA20-POLY1305:ECDHE-RSA-CHACHA20-POLY1305;
ssl_prefer_server_ciphers off;  # 让客户端选择更强的套件

# 防止 BEAST/POODLE 攻击
ssl_session_tickets off;

# HSTS（防中间人攻击）
add_header Strict-Transport-Security "max-age=63072000; includeSubDomains; preload" always;

# 其他安全响应头
add_header X-Content-Type-Options    "nosniff" always;
add_header X-Frame-Options           "DENY" always;  # 防点击劫持
add_header X-XSS-Protection          "1; mode=block" always;
add_header Referrer-Policy           "no-referrer-when-downgrade" always;
add_header Content-Security-Policy   "default-src 'self'" always;
add_header Permissions-Policy        "camera=(), microphone=()" always;
```

### 3.4 防止常见攻击

```nginx
# 防止路径遍历攻击
location ~* \.\. {
    return 403;
}

# 隐藏 .git 等敏感目录
location ~ /\. {
    deny all;
    access_log off;
    log_not_found off;
}

# 禁止执行上传目录中的脚本（防止 webshell）
location /uploads/ {
    location ~ \.(php|asp|aspx|jsp|cgi)$ {
        return 403;
    }
}

# 限制 HTTP 方法（只允许 GET/POST/HEAD）
if ($request_method !~ ^(GET|HEAD|POST)$) {
    return 405;
}

# 过滤恶意 User-Agent（爬虫/扫描器）
if ($http_user_agent ~* (sqlmap|nikto|nmap|metasploit|masscan)) {
    return 403;
}
```

## 四、性能调优综合配置

```nginx
worker_processes  auto;
worker_cpu_affinity auto;          # Worker 绑定 CPU 核（减少缓存失效）
worker_rlimit_nofile 65535;

events {
    worker_connections  65535;
    use epoll;
    multi_accept on;
    accept_mutex off;              # 高并发下关闭（减少竞争开销，内核已有 EPOLLEXCLUSIVE）
}

http {
    # 基础性能
    sendfile        on;
    tcp_nopush      on;
    tcp_nodelay     on;
    
    # 连接管理
    keepalive_timeout     75s;
    keepalive_requests    1000;    # 单个 Keep-Alive 最大复用请求数
    reset_timedout_connection on;  # 超时连接直接 RST（快速释放资源）
    
    # 缓冲区（根据实际响应大小调整）
    open_file_cache max=200000 inactive=20s;   # 文件描述符缓存
    open_file_cache_valid    30s;
    open_file_cache_min_uses 2;
    open_file_cache_errors   on;
    
    # 哈希表大小（避免频繁扩容）
    types_hash_max_size        2048;
    server_names_hash_max_size 512;
    server_names_hash_bucket_size 128;
    
    # 隐藏版本
    server_tokens off;
}
```

### 4.1 多核 CPU 绑定

```nginx
# 4 核机器的 CPU 亲和性配置
worker_processes 4;
worker_cpu_affinity 0001 0010 0100 1000;  # 手动绑定
# 或
worker_cpu_affinity auto;  # 自动绑定（推荐）
```

**为什么要 CPU 绑定？**  
每个 Worker 绑定到固定 CPU 核后，该核的 L1/L2 缓存（Worker 的套接字状态、内存池数据）命中率大幅提升，减少缓存行失效（cache line miss）导致的性能损耗。

### 4.2 open_file_cache 文件描述符缓存

Nginx 每次处理静态资源都需要 `open()`/`stat()` 系统调用获取文件信息，`open_file_cache` 将文件 fd 和 stat 信息缓存在内存中：

```nginx
open_file_cache max=200000 inactive=20s;  # 最多缓存 20 万个 fd，20s 不使用则过期
open_file_cache_valid    30s;             # 每 30s 验证一次缓存有效性
open_file_cache_min_uses 2;              # 文件至少被访问 2 次才缓存
open_file_cache_errors   on;             # 缓存文件不存在的信息（避免重复 open）
```

**效果**：对于静态资源服务器，`open_file_cache` 可以减少 40%+ 的系统调用，显著提升 QPS。
