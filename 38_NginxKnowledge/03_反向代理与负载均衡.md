## 一、反向代理原理

### 1.1 正向代理 vs 反向代理

```
正向代理（代理客户端）:
  客户端 ──→ 正向代理 ──→ 互联网目标
  客户端知道代理的存在；服务端看不到真实客户端

反向代理（代理服务端）:
  客户端 ──→ 反向代理(Nginx) ──→ 后端服务集群
  客户端不知道后端服务的存在；后端看不到真实客户端 IP（需配置 X-Real-IP 传递）
```

### 1.2 proxy_pass 指令

```nginx
server {
    # ① 不带斜杠：将整个 location 路径转发
    location /api/ {
        proxy_pass http://backend;
        # 请求 /api/users → 转发 /api/users
    }
    
    # ② 带斜杠：去掉 location 前缀后转发（URI 重写）
    location /api/ {
        proxy_pass http://backend/;
        # 请求 /api/users → 转发 /users（去掉了 /api/）
    }
    
    # ③ 带具体路径：替换 location 前缀
    location /api/ {
        proxy_pass http://backend/v2/;
        # 请求 /api/users → 转发 /v2/users
    }
}
```

### 1.3 必要的代理头部配置

```nginx
location / {
    proxy_pass http://backend;
    
    # 传递真实客户端信息
    proxy_set_header Host              $host;               # 保留原始 Host
    proxy_set_header X-Real-IP         $remote_addr;        # 真实客户端 IP
    proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;  # IP 链
    proxy_set_header X-Forwarded-Proto $scheme;             # http/https
    
    # 支持 WebSocket
    proxy_http_version 1.1;
    proxy_set_header Upgrade    $http_upgrade;
    proxy_set_header Connection $connection_upgrade;
    
    # 超时配置
    proxy_connect_timeout 5s;
    proxy_read_timeout    60s;
    proxy_send_timeout    60s;
}
```

### 1.4 代理缓存配置

```nginx
http {
    # 定义缓存路径（levels 两级目录/keys_zone 共享内存/inactive 不活跃时间/max_size 磁盘上限）
    proxy_cache_path /var/cache/nginx levels=1:2 keys_zone=api_cache:10m inactive=60m max_size=1g;
    
    server {
        location /api/static/ {
            proxy_pass http://backend;
            
            proxy_cache           api_cache;
            proxy_cache_key       "$scheme$request_method$host$request_uri";
            proxy_cache_valid     200 10m;   # 200 响应缓存 10 分钟
            proxy_cache_valid     404 1m;    # 404 响应缓存 1 分钟
            proxy_cache_use_stale error timeout updating;  # 上游故障时使用旧缓存
            
            add_header X-Cache-Status $upstream_cache_status;  # 调试用：HIT/MISS/BYPASS
        }
        
        location /api/dynamic/ {
            proxy_pass http://backend;
            proxy_no_cache 1;       # 不缓存
            proxy_cache_bypass 1;   # 不使用缓存
        }
    }
}
```

## 二、upstream 模块与负载均衡

### 2.1 五种负载均衡策略

```nginx
# ① 轮询（Round Robin，默认）─ 按顺序轮流分发
upstream rr_backend {
    server 127.0.0.1:8080;
    server 127.0.0.1:8081;
}

# ② 加权轮询（Weighted Round Robin）─ 按权重比分发
upstream weighted_backend {
    server 127.0.0.1:8080 weight=5;   # 分 5/8 的请求
    server 127.0.0.1:8081 weight=3;   # 分 3/8 的请求
}

# ③ IP Hash ─ 同一客户端 IP 总分发到同一后端（会话保持）
upstream iphash_backend {
    ip_hash;
    server 127.0.0.1:8080;
    server 127.0.0.1:8081;
    # 注意：若某节点下线，要用 down 标记而非直接删除（避免 hash 剧烈变动）
}

# ④ 最少连接（Least Connections）─ 优先分发到当前活动连接最少的后端
upstream leastconn_backend {
    least_conn;
    server 127.0.0.1:8080;
    server 127.0.0.1:8081;
}

# ⑤ 哈希（一致性哈希，需 ngx_http_upstream_hash_module）
upstream hash_backend {
    hash $request_uri consistent;  # 按 URI 哈希，consistent 启用一致性哈希
    server 127.0.0.1:8080;
    server 127.0.0.1:8081;
}
```

**选型建议**：
| 策略 | 适用场景 | 注意事项 |
|------|---------|---------|
| 轮询 | 无状态服务（主流微服务） | 不保证 Session 一致性 |
| 加权轮询 | 服务器性能不均 | 权重需根据实际压测调整 |
| IP Hash | 需要会话保持但无 Redis/Session | IP 变化（NAT/移动端）会导致会话丢失 |
| 最少连接 | 长连接、响应时间差异大 | 性能略低于轮询（需维护连接计数） |
| 哈希 | 缓存穿透优化 | consistent 模式下扩缩容影响最小 |

### 2.2 upstream 服务器参数

```nginx
upstream backend {
    server 127.0.0.1:8080 weight=5 max_fails=3 fail_timeout=30s;
    server 127.0.0.1:8081 weight=3 max_fails=3 fail_timeout=30s;
    server 127.0.0.1:8082 backup;   # 备用服务器（主服务器全部不可用时启用）
    server 127.0.0.1:8083 down;     # 标记为不可用（保留 ip_hash 中的位置）
    
    # upstream keepalive（与后端保持长连接池）
    keepalive 32;          # 连接池大小（每个 Worker 保持 32 个空闲连接）
    keepalive_timeout 60s; # 空闲连接超时
}
```

**参数说明**：
- `max_fails=3`：30s 内失败 3 次后将服务器标记为不可用
- `fail_timeout=30s`：被标记为不可用后，30s 后再次尝试
- `backup`：备用服务器，仅在所有主服务器不可用时接收流量
- `down`：永久标记为不可用（手动维护时使用）

### 2.3 被动健康检查 vs 主动健康检查

```nginx
# 被动健康检查（开源版内置）
# Nginx 通过监控请求失败次数来判断服务器是否健康
upstream backend {
    server 127.0.0.1:8080 max_fails=3 fail_timeout=30s;
}

# 主动健康检查（需要 nginx_upstream_check_module 或 Nginx Plus）
# 开源替代方案：使用 lua-resty-healthcheck 模块
upstream backend {
    server 127.0.0.1:8080;
    check interval=3000 rise=2 fall=3 timeout=1000 type=http;
    check_http_send "HEAD /health HTTP/1.0\r\n\r\n";
    check_http_expect_alive http_2xx http_3xx;
}
```

## 三、高级代理配置

### 3.1 灰度发布（金丝雀部署）

```nginx
# 方案一：基于请求头（推荐，可精确控制）
upstream prod_backend {
    server 192.168.1.10:8080;
    server 192.168.1.11:8080;
}

upstream canary_backend {
    server 192.168.1.20:8080;  # 新版本
}

map $http_x_canary $backend_pool {
    default     prod_backend;
    "true"      canary_backend;
}

server {
    location / {
        proxy_pass http://$backend_pool;
    }
}

# 方案二：基于 Cookie
map $cookie_canary $backend_pool {
    default  prod_backend;
    "1"      canary_backend;
}

# 方案三：基于权重（5% 流量到新版本）
upstream weighted_canary {
    server 192.168.1.10:8080 weight=95;
    server 192.168.1.20:8080 weight=5;
}
```

### 3.2 动静分离

```nginx
server {
    listen 80;
    server_name www.example.com;
    
    # 静态资源：直接从本地磁盘提供
    location ~* \.(html|css|js|jpg|png|gif|ico|woff|woff2|ttf|svg)$ {
        root /var/www/static;
        expires 30d;
        add_header Cache-Control "public, immutable";
        access_log off;  # 静态资源不记录访问日志，减少 I/O
    }
    
    # 动态请求：转发到后端应用
    location /api/ {
        proxy_pass http://backend;
    }
    
    # 前端 SPA（单页应用）
    location / {
        root /var/www/dist;
        try_files $uri $uri/ /index.html;  # history 模式路由支持
    }
}
```

### 3.3 WebSocket 代理

```nginx
http {
    # 根据 Upgrade 头决定 Connection 的值
    map $http_upgrade $connection_upgrade {
        default  upgrade;
        ''       close;
    }
    
    server {
        location /ws/ {
            proxy_pass http://websocket_backend;
            proxy_http_version 1.1;
            proxy_set_header Upgrade $http_upgrade;
            proxy_set_header Connection $connection_upgrade;
            proxy_set_header Host $host;
            proxy_read_timeout 3600s;   # WebSocket 长连接，需要大超时
            proxy_send_timeout 3600s;
        }
    }
}
```

### 3.4 CORS 跨域配置

```nginx
server {
    location /api/ {
        proxy_pass http://backend;
        
        # 添加 CORS 头
        add_header Access-Control-Allow-Origin  $http_origin always;
        add_header Access-Control-Allow-Methods 'GET, POST, PUT, DELETE, OPTIONS' always;
        add_header Access-Control-Allow-Headers 'Authorization, Content-Type, X-Requested-With' always;
        add_header Access-Control-Allow-Credentials true always;
        add_header Access-Control-Max-Age 86400 always;  # 预检请求缓存 1 天
        
        # 处理 OPTIONS 预检请求
        if ($request_method = 'OPTIONS') {
            return 204;
        }
    }
}
```

## 四、upstream 连接优化

### 4.1 长连接复用

```nginx
upstream backend {
    server 127.0.0.1:8080;
    keepalive 64;              # 每个 Worker 保持 64 个空闲长连接
    keepalive_requests 1000;   # 单个长连接最大请求数（防止连接老化）
    keepalive_timeout 60s;
}

server {
    location / {
        proxy_pass http://backend;
        proxy_http_version 1.1;                    # HTTP/1.1 才支持 keepalive
        proxy_set_header Connection "";            # 清除 Connection 头（否则默认 close）
    }
}
```

**为什么要设置 `Connection ""`？**  
客户端请求中可能包含 `Connection: close`，如果直接透传给后端，会导致每次请求都断开连接，无法复用长连接池。清除 Connection 头后，Nginx 与后端之间始终使用 HTTP/1.1 长连接。
