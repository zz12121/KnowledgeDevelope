## 1. 向量搜索（Vector Search）

### 1.1 基本概念

**向量搜索（语义搜索）** 通过将文本/图片转换为高维向量（Embedding），计算向量间的相似度（余弦/点积/欧氏距离）来找到语义相近的内容，突破关键字匹配的局限。

**与传统关键字搜索对比：**

| 维度 | 关键字搜索 | 向量搜索 |
|------|-----------|---------|
| 原理 | 倒排索引，词项匹配 | 近似最近邻（ANN），向量距离 |
| 优势 | 精确匹配，速度快 | 语义理解，同义词/近义词 |
| 劣势 | 无法理解语义 | 需要 Embedding 模型，资源消耗大 |
| 适用 | 精确查询、过滤 | 语义搜索、推荐、问答 |

### 1.2 Dense Vector 配置

```json
PUT /articles
{
  "mappings": {
    "properties": {
      "title": {"type": "text", "analyzer": "ik_max_word"},
      "content": {"type": "text"},
      "embedding": {
        "type": "dense_vector",
        "dims": 768,              // 向量维度（与模型输出一致）
        "index": true,            // true: 支持 kNN 查询；false: 只存储
        "similarity": "cosine",   // 相似度度量：cosine / dot_product / l2_norm
        "index_options": {
          "type": "hnsw",         // 近似 ANN 索引算法（Hierarchical Navigable Small World）
          "m": 16,                // 图连接数（越大精度越高，资源越多）
          "ef_construction": 100  // 建图时动态候选集大小
        }
      }
    }
  }
}
```

## 2. Ingest Pipeline 详解

**Ingest Pipeline 是 ES 内置的 ETL 引擎**，在文档写入索引前执行预处理，无需外部 Logstash。

### 2.1 常用 Processor

```json
PUT /_ingest/pipeline/full-pipeline
{
  "description": "完整数据处理 Pipeline",
  "processors": [
    // 1. Grok：正则解析
    {
      "grok": {
        "field": "log_line",
        "patterns": ["%{TIMESTAMP_ISO8601:timestamp} %{LOGLEVEL:level} %{GREEDYDATA:message}"],
        "ignore_failure": true
      }
    },
    // 2. Date：日期解析和格式化
    {
      "date": {
        "field": "timestamp",
        "target_field": "@timestamp",
        "formats": ["ISO8601", "yyyy-MM-dd HH:mm:ss"],
        "timezone": "Asia/Shanghai"
      }
    },
    // 3. GeoIP：IP 地址转地理位置
    {
      "geoip": {
        "field": "client_ip",
        "target_field": "geo",
        "properties": ["country_name", "city_name", "location"]
      }
    },
    // 4. User Agent：解析 UA 字符串
    {
      "user_agent": {
        "field": "user_agent_string",
        "target_field": "user_agent"
      }
    },
    // 5. Set：设置固定值或基于脚本的值
    {
      "set": {
        "field": "ingest_time",
        "value": "{{_ingest.timestamp}}"
      }
    },
    // 6. Rename：重命名字段
    {
      "rename": {"field": "old_name", "target_field": "new_name", "ignore_missing": true}
    },
    // 7. Remove：删除字段
    {
      "remove": {"field": ["log_line", "timestamp", "user_agent_string"]}
    },
    // 8. Convert：类型转换
    {
      "convert": {"field": "status_code", "type": "integer"}
    },
    // 9. Script：自定义脚本处理
    {
      "script": {
        "source": """
          if (ctx.price != null && ctx.quantity != null) {
            ctx.total = ctx.price * ctx.quantity;
          }
        """
      }
    },
    // 10. Foreach：处理数组字段
    {
      "foreach": {
        "field": "tags",
        "processor": {
          "uppercase": {"field": "_ingest._value"}
        }
      }
    }
  ],
  "on_failure": [
    {
      "set": {
        "field": "_index",
        "value": "failed-{{_index}}"
      }
    }
  ]
}
```

```java
// 实体类
@Document(indexName = "products")
@Setting(settingPath = "elasticsearch/product-settings.json")
public class ProductDocument {
    
    @Id
    private String id;
    
    @Field(type = FieldType.Text, analyzer = "ik_max_word", searchAnalyzer = "ik_smart")
    private String title;
    
    @MultiField(mainField = @Field(type = FieldType.Text, analyzer = "ik_max_word"),
                otherFields = {@InnerField(suffix = "keyword", type = FieldType.Keyword)})
    private String description;
    
    @Field(type = FieldType.Keyword)
    private String brand;
    
    @Field(type = FieldType.Double)
    private Double price;
    
    @Field(type = FieldType.Integer)
    private Integer salesCount;
    
    @Field(type = FieldType.Boolean)
    private Boolean isActive;
    
    @Field(type = FieldType.Keyword)
    private List<String> tags;
    
    @Field(type = FieldType.Date, format = DateFormat.date_time)
    private LocalDateTime createdAt;
}

// Repository
public interface ProductRepository extends ElasticsearchRepository<ProductDocument, String> {
    
    List<ProductDocument> findByBrand(String brand);
    
    Page<ProductDocument> findByIsActiveTrue(Pageable pageable);
    
    @Query("{\"bool\": {\"must\": [{\"match\": {\"title\": \"?0\"}}], \"filter\": [{\"range\": {\"price\": {\"gte\": \"?1\", \"lte\": \"?2\"}}}]}}")
    List<ProductDocument> searchByTitleAndPriceRange(String keyword, Double minPrice, Double maxPrice);
}

// Service 层（复杂查询使用 ElasticsearchOperations）
@Service
@RequiredArgsConstructor
public class ProductSearchService {
    
    private final ElasticsearchOperations esOps;
    private final ProductRepository productRepository;
    
    /**
     * 综合搜索（关键字 + 品牌过滤 + 价格范围 + 排序 + 分页）
     */
    public SearchResult search(SearchRequest request) {
        // 构建查询
        BoolQueryBuilder boolQuery = QueryBuilders.boolQuery();
        
        // 全文搜索（query context）
        if (StringUtils.hasText(request.getKeyword())) {
            boolQuery.must(QueryBuilders.multiMatchQuery(request.getKeyword(), "title^3", "description^1"));
        }
        
        // 过滤条件（filter context，可缓存）
        boolQuery.filter(QueryBuilders.termQuery("is_active", true));
        if (StringUtils.hasText(request.getBrand())) {
            boolQuery.filter(QueryBuilders.termQuery("brand", request.getBrand()));
        }
        if (request.getMinPrice() != null || request.getMaxPrice() != null) {
            RangeQueryBuilder rangeQuery = QueryBuilders.rangeQuery("price");
            if (request.getMinPrice() != null) rangeQuery.gte(request.getMinPrice());
            if (request.getMaxPrice() != null) rangeQuery.lte(request.getMaxPrice());
            boolQuery.filter(rangeQuery);
        }
        
        // 排序
        SortBuilder<?> sort = switch (request.getSortBy()) {
            case "price_asc" -> SortBuilders.fieldSort("price").order(SortOrder.ASC);
            case "price_desc" -> SortBuilders.fieldSort("price").order(SortOrder.DESC);
            case "sales" -> SortBuilders.fieldSort("salesCount").order(SortOrder.DESC);
            default -> SortBuilders.scoreSort();
        };
        
        // 聚合（左侧筛选栏）
        TermsAggregationBuilder brandAgg = AggregationBuilders.terms("brands").field("brand").size(20);
        RangeAggregationBuilder priceAgg = AggregationBuilders.range("price_ranges")
            .field("price")
            .addRange(0, 1000).addRange(1000, 3000).addRange(3000, 5000).addUnboundedFrom(5000);
        
        // 构建 Query
        NativeSearchQuery query = new NativeSearchQueryBuilder()
            .withQuery(boolQuery)
            .withSorts(sort)
            .withPageable(PageRequest.of(request.getPage(), request.getPageSize()))
            .withAggregations(brandAgg, priceAgg)
            .withHighlightFields(new HighlightBuilder.Field("title").numOfFragments(0),
                                 new HighlightBuilder.Field("description").numOfFragments(2).fragmentSize(150))
            .build();
        
        SearchHits<ProductDocument> hits = esOps.search(query, ProductDocument.class);
        
        // 处理结果
        List<ProductVO> products = hits.getSearchHits().stream()
            .map(hit -> {
                ProductVO vo = ProductVO.from(hit.getContent());
                // 高亮处理
                if (hit.getHighlightFields().containsKey("title")) {
                    vo.setTitle(hit.getHighlightFields().get("title").get(0));
                }
                return vo;
            })
            .collect(Collectors.toList());
        
        // 处理聚合结果
        Map<String, Long> brandFacets = new LinkedHashMap<>();
        if (hits.getAggregations() != null) {
            Terms brandsAgg = hits.getAggregations().get("brands");
            brandsAgg.getBuckets().forEach(b -> brandFacets.put(b.getKeyAsString(), b.getDocCount()));
        }
        
        return SearchResult.builder()
            .products(products)
            .total(hits.getTotalHits())
            .brandFacets(brandFacets)
            .build();
    }
    
    /**
     * 批量写入（高效）
     */
    public void bulkIndex(List<ProductDocument> products) {
        productRepository.saveAll(products);
    }
    
    /**
     * search_after 深分页
     */
    public List<ProductDocument> scrollSearch(Object[] searchAfter) {
        NativeSearchQueryBuilder queryBuilder = new NativeSearchQueryBuilder()
            .withQuery(QueryBuilders.matchAllQuery())
            .withSorts(SortBuilders.fieldSort("createdAt").order(SortOrder.DESC),
                       SortBuilders.fieldSort("_id").order(SortOrder.ASC))
            .withPageable(PageRequest.of(0, 100));
        
        if (searchAfter != null) {
            queryBuilder.withSearchAfter(searchAfter);
        }
        
        return esOps.search(queryBuilder.build(), ProductDocument.class)
            .getSearchHits().stream()
            .map(SearchHit::getContent)
            .collect(Collectors.toList());
    }
}
```

## 6. Kafka 与 ES 集成

### 6.1 Kafka → ES 数据流

```java
@Configuration
public class KafkaToEsConfig {
    
    @Bean
    public ConcurrentKafkaListenerContainerFactory<String, String> kafkaListenerContainerFactory() {
        ConcurrentKafkaListenerContainerFactory<String, String> factory =
            new ConcurrentKafkaListenerContainerFactory<>();
        factory.getContainerProperties().setAckMode(AckMode.MANUAL_IMMEDIATE);
        factory.setBatchListener(true);  // 批量消费
        factory.setConcurrency(4);       // 4个消费者线程
        return factory;
    }
}

@Service
@RequiredArgsConstructor
public class KafkaToEsConsumer {
    
    private final ElasticsearchClient esClient;
    private final ObjectMapper objectMapper;
    
    @KafkaListener(topics = "product-events", groupId = "es-writer", containerFactory = "kafkaListenerContainerFactory")
    public void consume(List<ConsumerRecord<String, String>> records, Acknowledgment ack) {
        try {
            BulkRequest.Builder bulk = new BulkRequest.Builder();
            
            for (ConsumerRecord<String, String> record : records) {
                ProductEvent event = objectMapper.readValue(record.value(), ProductEvent.class);
                
                switch (event.getType()) {
                    case "CREATE", "UPDATE" ->
                        bulk.operations(op -> op.index(i -> i
                            .index("products")
                            .id(event.getId())
                            .document(event.getProduct())));
                    case "DELETE" ->
                        bulk.operations(op -> op.delete(d -> d
                            .index("products")
                            .id(event.getId())));
                }
            }
            
            BulkResponse response = esClient.bulk(bulk.build());
            
            if (response.errors()) {
                // 处理部分失败（写日志，发告警）
                response.items().stream()
                    .filter(item -> item.error() != null)
                    .forEach(item -> log.error("ES write failed: {}", item.error().reason()));
            }
            
            ack.acknowledge();  // 手动提交 offset
            
        } catch (Exception e) {
            log.error("Kafka to ES sync failed", e);
            // 不 ACK，Kafka 会重新投递
        }
    }
}
```
