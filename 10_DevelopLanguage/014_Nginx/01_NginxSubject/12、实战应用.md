###### 1. 如何使用 Nginx 搭建静态网站？

```nginx
server {
    listen 80;
    server_name www.example.com example.com;
    root /var/www/html;
    index index.html index.htm;
    
    # SPA（React/Vue）：所有路径都走 index.html
    location / {
        try_files $uri $uri/ /index.html;
    }
    
    # 静态资源缓存
    location ~* \.(css|js|png|jpg|gif|ico|svg|woff2|ttf)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
        access_log off;
        gzip_static on;   # 使用预压缩文件（.gz）
    }
    
    # 禁止访问隐藏文件（.git/.htaccess）
    location ~ /\. {
        deny all;
    }
    
    # 自定义错误页
    error_page 404 /404.html;
    error_page 500 502 503 504 /50x.html;
    location = /50x.html {
        root /usr/share/nginx/html;
    }
}
```

###### 2. 如何使用 Nginx 实现灰度发布？

灰度发布（Canary Release）让小比例用户访问新版本，验证无误后全量推送：

**方案一：基于 Cookie 的灰度（最常用）**
```nginx
upstream stable {
    server 192.168.1.10:8080;
    server 192.168.1.11:8080;
}

upstream canary {
    server 192.168.1.20:8080;   # 新版本
}

map $http_cookie $pool {
    default         stable;
    "~*canary=true" canary;     # Cookie 中含 canary=true 的用户走新版本
}

server {
    location /api/ {
        proxy_pass http://$pool;
    }
}
```

**方案二：基于请求比例（Nginx Plus 或 Lua）**
```nginx
# 使用 Lua + 随机数控制流量比例
location /api/ {
    access_by_lua_block {
        math.randomseed(ngx.now())
        if math.random() < 0.1 then   -- 10% 流量到金丝雀
            ngx.var.upstream = "canary"
        else
            ngx.var.upstream = "stable"
        end
    }
    proxy_pass http://$upstream;
}
```

**方案三：基于 Header**
```nginx
map $http_x_canary $pool {
    "1"     canary;
    default stable;
}
# 测试时在请求头加 X-Canary: 1 即可访问新版本
```

###### 3. 如何使用 Nginx 实现 A/B 测试？

A/B 测试与灰度发布思路类似，区别在于 A/B 测试是**固定某个用户始终访问同一版本**（避免体验割裂）：

```nginx
# 基于用户 ID 哈希，保证同一用户始终访问同一版本
map $cookie_userid $ab_group {
    default "A";
    # 奇数用户 → B 组（~50%）
    ~*[13579]$  "B";
}

upstream group_a {
    server 192.168.1.10:8080;
}

upstream group_b {
    server 192.168.1.20:8080;
}

server {
    location /api/ {
        if ($ab_group = "B") {
            proxy_pass http://group_b;
            break;
        }
        proxy_pass http://group_a;
    }
    
    # 在响应头中标注当前组别，方便埋点统计
    add_header X-AB-Group $ab_group;
}
```

###### 4. 如何使用 Nginx 配置跨域请求（CORS）？

```nginx
server {
    location /api/ {
        # 处理预检请求（OPTIONS）
        if ($request_method = 'OPTIONS') {
            add_header Access-Control-Allow-Origin  $http_origin always;
            add_header Access-Control-Allow-Methods 'GET, POST, PUT, DELETE, OPTIONS' always;
            add_header Access-Control-Allow-Headers 'Authorization, Content-Type, X-Requested-With' always;
            add_header Access-Control-Allow-Credentials 'true' always;
            add_header Access-Control-Max-Age 3600;     # 预检缓存 1 小时
            add_header Content-Length 0;
            return 204;
        }
        
        # 正常请求的 CORS 响应头
        add_header Access-Control-Allow-Origin  $http_origin always;
        add_header Access-Control-Allow-Methods 'GET, POST, PUT, DELETE, OPTIONS' always;
        add_header Access-Control-Allow-Headers 'Authorization, Content-Type, X-Requested-With' always;
        add_header Access-Control-Allow-Credentials 'true' always;
        
        proxy_pass http://backend;
    }
}
```

**安全注意事项：**
- 不要写 `Access-Control-Allow-Origin: *`，这会允许所有域，且无法配合 `Allow-Credentials: true`
- 用 `$http_origin` 配合白名单校验（使用 `map` 或 Lua 实现）：
```nginx
map $http_origin $cors_origin {
    "https://www.example.com"    $http_origin;
    "https://app.example.com"    $http_origin;
    default                      "";
}
add_header Access-Control-Allow-Origin $cors_origin always;
```

###### 5. 如何使用 Nginx 配置 WebSocket？

WebSocket 需要 HTTP 升级（Upgrade），Nginx 默认不会转发 `Connection: Upgrade` 头：

```nginx
upstream ws_backend {
    server 127.0.0.1:8080;
    keepalive 10;
}

server {
    listen 80;
    server_name ws.example.com;
    
    location /ws/ {
        proxy_pass http://ws_backend;
        
        # 关键配置：支持 HTTP → WebSocket 协议升级
        proxy_http_version 1.1;
        proxy_set_header Upgrade    $http_upgrade;    # 传递 Upgrade 头
        proxy_set_header Connection "upgrade";        # 设置 Connection: upgrade
        
        proxy_set_header Host               $host;
        proxy_set_header X-Real-IP          $remote_addr;
        proxy_set_header X-Forwarded-For    $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto  $scheme;
        
        # WebSocket 连接通常是长连接，超时要设置得够长
        proxy_read_timeout  3600s;    # 1 小时
        proxy_send_timeout  3600s;
        proxy_connect_timeout 10s;
    }
    
    # 普通 HTTP 请求
    location / {
        proxy_pass http://ws_backend;
        proxy_http_version 1.1;
        proxy_set_header Connection "";
    }
}
```

###### 6. 如何使用 Nginx 实现正向代理？

正向代理让内网机器通过 Nginx 访问外部网络（需要 `ngx_http_proxy_connect_module` 支持 HTTPS 代理）：

```nginx
server {
    listen 3128;    # 正向代理端口
    
    # HTTP 正向代理（内置支持）
    location / {
        resolver 8.8.8.8;                # DNS 解析器（必须配置）
        proxy_pass $scheme://$http_host$request_uri;
        proxy_set_header Host $http_host;
        proxy_connect_timeout 10s;
        proxy_read_timeout    60s;
    }
}
```

**注意：** Nginx 原生**不支持 HTTPS（CONNECT 方法）正向代理**，需要编译 `ngx_http_proxy_connect_module` 模块。正向代理场景通常用 Squid 或 3proxy 更合适。

###### 7. 如何使用 Nginx 配置多个域名？

```nginx
# 方案一：多个 server 块（完全独立配置）
server {
    listen 443 ssl http2;
    server_name api.example.com;
    ssl_certificate /etc/nginx/ssl/api.example.com.pem;
    ssl_certificate_key /etc/nginx/ssl/api.example.com.key;
    location / { proxy_pass http://api-backend; }
}

server {
    listen 443 ssl http2;
    server_name admin.example.com;
    ssl_certificate /etc/nginx/ssl/admin.example.com.pem;
    ssl_certificate_key /etc/nginx/ssl/admin.example.com.key;
    location / { proxy_pass http://admin-backend; }
}

# 方案二：泛域名证书 + 一个 server 块 + map 路由
server {
    listen 443 ssl http2;
    server_name *.example.com example.com;
    ssl_certificate /etc/nginx/ssl/wildcard.example.com.pem;   # 泛域名证书
    ssl_certificate_key /etc/nginx/ssl/wildcard.example.com.key;
    
    map $host $upstream_server {
        api.example.com     http://api-backend;
        admin.example.com   http://admin-backend;
        www.example.com     http://web-backend;
        default             http://web-backend;
    }
    
    location / {
        proxy_pass $upstream_server;
    }
}
```

###### 8. 如何使用 Nginx 实现动静分离？

动静分离让 Nginx 直接处理静态资源，动态请求转发给应用服务器：

```nginx
server {
    listen 80;
    server_name www.example.com;
    
    # 静态资源：Nginx 直接处理
    location ~* \.(html|css|js|jpg|jpeg|png|gif|ico|svg|woff|woff2|ttf|eot)$ {
        root /var/www/static;
        expires 30d;
        add_header Cache-Control "public, immutable";
        access_log off;
        gzip on;
        try_files $uri =404;
    }
    
    # 动态请求：转发给 Spring Boot
    location /api/ {
        proxy_pass http://127.0.0.1:8080;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }
    
    # 其他请求走前端 SPA
    location / {
        root /var/www/html;
        try_files $uri $uri/ /index.html;
    }
}
```

**效果：** 静态资源由 Nginx 直接响应，不经过 JVM，响应速度从 100ms 降到 1ms，同时大幅降低 Tomcat/Spring Boot 的负载。

###### 9. 如何使用 Nginx 配置图片服务器？

```nginx
server {
    listen 80;
    server_name img.example.com;
    
    root /var/www/images;
    
    # 图片访问
    location ~* \.(jpg|jpeg|png|gif|webp|svg|ico)$ {
        expires 365d;
        add_header Cache-Control "public, immutable";
        add_header Vary Accept;                    # 支持 WebP 协商
        
        # 防盗链
        valid_referers none blocked *.example.com;
        if ($invalid_referer) {
            return 403;
        }
        
        try_files $uri =404;
    }
    
    # 限制上传接口（如果 Nginx 作为上传代理）
    location /upload/ {
        client_max_body_size  50m;   # 图片最大 50MB
        proxy_pass http://upload-backend;
    }
    
    # 禁止目录浏览
    autoindex off;
    
    # 限制下载速度（防止带宽被占满）
    location /large/ {
        limit_rate 1m;    # 每秒最多 1MB
    }
}
```

###### 10. 如何使用 Nginx 实现文件下载限速？

```nginx
server {
    location /download/ {
        root /var/www/files;
        
        # 方法一：全局限速
        limit_rate 512k;        # 限制下载速度为 512KB/s
        
        # 方法二：下载前 N 字节不限速，之后再限速（改善下载初体验）
        limit_rate_after 5m;    # 前 5MB 全速，之后限制为 512KB/s
        limit_rate 512k;
        
        # 方法三：根据请求参数动态限速（需要 Lua）
        # access_by_lua_block {
        #     local rate = ngx.var.arg_speed or "512k"
        #     ngx.var.limit_rate = rate
        # }
    }
}
```

###### 追问：Nginx 如何实现基于 IP 的流量分区（按地域路由到不同后端）？

```nginx
# 使用 geo 模块按 IP 段分组
geo $region {
    default     "overseas";
    10.0.0.0/8  "internal";
    # 配合 MaxMind GeoIP 可以按国家划分，但需要 geoip 模块
    # 简单按 IP 段规划也很实用（内网/外网/不同 IDC）
}

upstream cn_backend     { server 192.168.1.10:8080; }
upstream us_backend     { server 10.20.1.10:8080; }
upstream internal_backend { server 172.16.1.10:8080; }

map $region $upstream_pool {
    "internal"  internal_backend;
    "overseas"  us_backend;
    default     cn_backend;
}

server {
    location /api/ {
        proxy_pass http://$upstream_pool;
    }
}
```
