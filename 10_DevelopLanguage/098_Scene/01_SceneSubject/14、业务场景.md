###### 1. 如何设计一个优惠券系统？

**核心数据模型：**

```sql
-- 优惠券模板（规则定义）
CREATE TABLE coupon_template (
    id BIGINT PRIMARY KEY,

    discount_amount DECIMAL(10,2) COMMENT '满减金额/折扣比例',

    end_time DATETIME,

CREATE TABLE user_coupon (
    id BIGINT PRIMARY KEY,

    user_id BIGINT,
    status TINYINT COMMENT '0未使用 1已使用 2已过期',
    received_time DATETIME,
    used_time DATETIME,
    order_id BIGINT COMMENT '使用时关联的订单ID',
    INDEX idx_user_status(user_id, status)

```

**领券流程（防超量）：**

```java
// Redis 控制发行量上限
public boolean receiveCoupon(Long userId, Long templateId) {
    String countKey = "coupon:count:" + templateId;
    String userKey = "coupon:received:" + templateId + ":" + userId;
    
    // Lua 脚本原子操作：检查用户是否已领 + 检查总量 + 扣减
    String lua = 
        "if redis.call('sismember', KEYS[1], ARGV[1]) == 1 then return -1 end " +
        "local count = tonumber(redis.call('get', KEYS[2])) " +
        "if count == nil or count <= 0 then return -2 end " +
        "redis.call('sadd', KEYS[1], ARGV[1]) " +
        "redis.call('decr', KEYS[2]) " +
        "return 1";
    
    Long result = redisTemplate.execute(new DefaultRedisScript<>(lua, Long.class),
        List.of(userKey, countKey), userId.toString());
    
    if (result == 1) {
        // 异步写 DB
        couponMapper.insertUserCoupon(userId, templateId);
        return true;
    }
    return false;
}
```

**优惠券抵扣计算：**

```java
public BigDecimal calculateDiscount(Order order, UserCoupon coupon) {
    CouponTemplate template = coupon.getTemplate();
    
    // 检查使用条件
    if (order.getTotalAmount().compareTo(template.getThreshold()) < 0) {
        throw new CouponNotApplicableException("未达到使用门槛");
    }
    
    return switch (template.getType()) {
        case FULL_REDUCTION -> template.getDiscountAmount(); // 满减
        case DISCOUNT -> order.getTotalAmount()
            .multiply(BigDecimal.ONE.subtract(template.getDiscountAmount())); // 折扣
        case DIRECT_REDUCTION -> template.getDiscountAmount(); // 直减
        default -> BigDecimal.ZERO;
    };
}
```

**防刷策略：** 每个用户每种优惠券限领N张；黑名单机制（同一设备/IP 批量领券）；优惠券与手机号/实名认证绑定。

###### 2. 如何设计一个积分系统？

积分系统的特点：**高频写（各种行为都产生积分）、精确（涉及资产，不能出错）、可审计（需要流水记录）**。

**数据模型：**

```sql
-- 积分账户
CREATE TABLE points_account (
    id BIGINT PRIMARY KEY,
    user_id BIGINT UNIQUE,
    total_points BIGINT DEFAULT 0 COMMENT '累计积分（不减）',
    available_points BIGINT DEFAULT 0 COMMENT '可用积分（可减）',
    frozen_points BIGINT DEFAULT 0 COMMENT '冻结积分',
    version INT DEFAULT 0 COMMENT '乐观锁版本号'
);

-- 积分流水（完整记录每次变动）
CREATE TABLE points_record (
    id BIGINT PRIMARY KEY,
    user_id BIGINT,
    change_amount BIGINT COMMENT '变动金额（正=增加，负=扣减）',
    balance BIGINT COMMENT '变动后余额',
    type VARCHAR(30) COMMENT '类型：ORDER_REWARD/SIGN_IN/EXCHANGE等',
    description VARCHAR(200),
    order_id BIGINT,
    expire_time DATETIME COMMENT '积分过期时间',
    create_time DATETIME
);
```

**积分变动（保证精确性）：**

```java
@Transactional
public void addPoints(Long userId, long amount, String type, String reason) {
    // 乐观锁更新，失败重试
    int retries = 3;
    while (retries-- > 0) {
        PointsAccount account = pointsMapper.selectByUserId(userId);
        int updated = pointsMapper.updatePoints(
            userId, amount, account.getVersion()); // WHERE version = #{version}
        if (updated > 0) {
            // 写流水
            PointsRecord record = new PointsRecord(userId, amount, 
                account.getAvailablePoints() + amount, type, reason);
            recordMapper.insert(record);
            return;
        }
    }
    throw new OptimisticLockException("积分更新冲突，请重试");
}
```

**积分过期处理：** 每条积分有过期时间，定时任务扫描过期积分，扣减账户（先进先出 FIFO——优先消费最早要过期的积分）。

**积分兑换防超兑：** 兑换前先冻结（available_points -= amount, frozen_points += amount），兑换成功后扣除冻结（frozen_points -= amount），失败则解冻。

@GlobalTransactional
public Order createOrder(CreateOrderRequest req) {
    // 1. 参数校验（商品存在、用户地址有效）
    // 2. 查商品价格（防止价格被篡改，以服务端价格为准）
    List<Product> products = productService.getByIds(req.getProductIds());
    
    // 3. 计算金额
    BigDecimal totalAmount = calculateTotal(products, req.getItems());
    BigDecimal discountAmount = calculateDiscount(req.getCouponId(), totalAmount);
    
    // 4. 预扣库存（库存服务）
    inventoryService.reserveStock(req.getItems()); // 分布式事务
    
    // 5. 冻结优惠券
    if (req.getCouponId() != null) {
        couponService.freezeCoupon(req.getCouponId(), req.getUserId());
    }
    
    // 6. 写订单到 DB
    Order order = buildOrder(req, products, totalAmount, discountAmount);
    orderMapper.insert(order);
    
    // 7. 发延迟消息（30分钟后检查是否支付）
    mqTemplate.sendDelay("order.timeout", order.getId(), 30, TimeUnit.MINUTES);
    
    return order;
}
```

@RocketMQMessageListener(topic = "order.timeout")
public void handleOrderTimeout(Long orderId) {
    Order order = orderMapper.selectById(orderId);
    if (order.getStatus() == OrderStatus.PENDING_PAYMENT) {
        // CAS 状态更新，防止并发（支付和超时同时到来）
        int rows = orderMapper.cancelIfPending(orderId);
        if (rows > 0) {
            inventoryService.releaseStock(orderId); // 释放库存
            couponService.releaseCoupon(order.getCouponId()); // 释放优惠券
        }
    }
}
```

###### 4. 如何设计一个库存系统？

库存系统最大的挑战：**高并发下的精确扣减，不超卖，不漏卖**。

**多级库存模型：**

```
总库存（WMS）
    ↓
可售库存（已扣除被预留的部分）
    ↓
预售库存（暂时不可用，等待转正）
```

**数据模型：**

```sql
CREATE TABLE inventory (
    id BIGINT PRIMARY KEY,
    sku_id BIGINT UNIQUE,
    total_stock INT COMMENT '总库存',
    available_stock INT COMMENT '可售库存（可以卖的）',
    frozen_stock INT COMMENT '已预占但未扣减的库存（下单已锁定）',
    version INT DEFAULT 0
);
```

**库存扣减策略（防超卖）：**

**预占 + 实扣两阶段：**

1. **下单预占（减可售库存，增冻结库存）：**
```sql
UPDATE inventory 
SET available_stock = available_stock - #{qty},
    frozen_stock = frozen_stock + #{qty},
    version = version + 1
WHERE sku_id = #{skuId} AND available_stock >= #{qty}
-- affected rows == 0 → 库存不足
```

2. **支付成功后实扣（减冻结库存）：**
```sql
UPDATE inventory 
SET frozen_stock = frozen_stock - #{qty},
    total_stock = total_stock - #{qty}
WHERE sku_id = #{skuId}
```

3. **取消订单释放（减冻结，增可售）：**
```sql
UPDATE inventory 
SET frozen_stock = frozen_stock - #{qty},
    available_stock = available_stock + #{qty}
WHERE sku_id = #{skuId}
```

**秒杀库存预热：**

活动开始前，将热门商品的库存预加载到 Redis，扣减走 Redis（Lua 原子操作），定时同步回 DB，避免 DB 承受秒杀洪峰。

**库存超卖最终兜底：**

即使 Redis 预扣成功，DB 落库时再加乐观锁校验，确保双重保险。

###### 5. 如何设计一个支付系统？

支付系统的核心要求：**高安全、幂等、高一致性、可审计**。

**整体流程：**

```
用户确认支付 → 支付服务创建支付单 → 调用第三方支付
              ↓                          ↓
           写支付单（PENDING）         异步回调（PAID/FAILED）
              ↓                          ↓
           轮询结果（超时保护）       支付服务处理回调
              ↓                          ↓
           更新支付单状态           通知订单服务（MQ）
```

**数据模型：**

```sql
CREATE TABLE payment (
    id BIGINT PRIMARY KEY,
    pay_no VARCHAR(50) UNIQUE COMMENT '支付单号（内部，幂等键）',
    order_id BIGINT,
    user_id BIGINT,
    amount DECIMAL(18,2),
    channel TINYINT COMMENT '1支付宝 2微信支付 3银联',
    status TINYINT COMMENT '0待支付 1支付中 2支付成功 3支付失败 4已退款',
    third_party_trade_no VARCHAR(100) COMMENT '第三方流水号',
    callback_content TEXT COMMENT '回调原文（重要，留存）',
    create_time DATETIME,
    pay_time DATETIME,
    expire_time DATETIME COMMENT '支付超时',
    UNIQUE KEY uk_order_channel(order_id, channel) COMMENT '同订单同渠道只能有一笔'
);
```

**回调幂等处理：**

```java
@PostMapping("/callback/alipay")
public String alipayCallback(@RequestParam Map<String, String> params) {
    // 1. 验证签名（防伪造回调）
    if (!alipayClient.verifySign(params)) {
        return "fail";
    }
    
    String payNo = params.get("out_trade_no");
    String tradeStatus = params.get("trade_status");
    
    if ("TRADE_SUCCESS".equals(tradeStatus)) {
        // 2. 幂等处理（CAS 状态更新，防重复回调）
        int rows = paymentMapper.updateStatusIfPending(payNo, PayStatus.SUCCESS);
        if (rows > 0) {
            // 3. 通知订单服务（MQ）
            mqTemplate.send("payment.success", new PaySuccessEvent(payNo));
        }
    }
    return "success";
}
```

**主动查单（对账）：**

回调不可靠，定时任务主动向第三方查询支付状态，处理回调遗漏：

```java
@Scheduled(fixedDelay = 60_000)
public void reconcile() {
    // 查找创建超过5分钟仍处于 PENDING 的支付单
    List<Payment> pendingPayments = paymentMapper.findLongPending(5);
    pendingPayments.forEach(p -> {
        PayQueryResult result = alipayClient.queryOrder(p.getPayNo());
        if (result.isPaid()) {
            handlePaySuccess(p, result);
        } else if (result.isExpired()) {
            paymentMapper.updateStatus(p.getId(), PayStatus.FAILED);
        }
    });
}
```

###### 6. 如何设计一个退款系统？

退款系统是支付的逆向流程，涉及资金退回，要求**安全、可审计、不重复退款**。

**退款状态机：**

```
退款申请 → 审核通过 → 调用第三方退款 → 退款成功/失败
         → 审核拒绝
```

**退款类型：** 全额退款、部分退款（退指定商品）、退差价（价保）。

**防重退款（核心安全控制）：**

```java
// 同一笔订单，不能重复退款超过支付金额
@Transactional
public void applyRefund(RefundRequest req) {
    Payment payment = paymentMapper.findByOrderId(req.getOrderId());
    
    // 已退款金额
    BigDecimal refundedAmount = refundMapper.sumRefundedAmount(req.getOrderId());
    
    // 申请退款金额 + 已退款金额 不能超过支付金额
    if (refundedAmount.add(req.getRefundAmount()).compareTo(payment.getAmount()) > 0) {
        throw new RefundExceededException("退款金额超过支付金额");
    }
    
    // 创建退款单（PENDING）
    Refund refund = new Refund(req.getOrderId(), req.getRefundAmount(), "PENDING");
    refundMapper.insert(refund);
    
    // 异步调用第三方退款
    mqTemplate.send("refund.process", refund.getId());
}
```

**第三方退款调用（幂等）：**

```java
// 以退款单号作为幂等键调用三方退款
public void processRefund(Long refundId) {
    Refund refund = refundMapper.selectById(refundId);
    
    try {
        AlipayRefundResult result = alipayClient.refund(
            refund.getOrderId().toString(),  // 原支付单号
            refund.getId().toString(),        // 退款请求号（幂等键）
            refund.getAmount()
        );
        
        if (result.isSuccess()) {
            refundMapper.updateStatus(refundId, "SUCCESS");
            inventoryService.addStock(refund.getOrderId()); // 恢复库存
            pointsService.deductPoints(refund.getOrderId()); // 扣回积分
        }
    } catch (Exception e) {
        refundMapper.updateStatus(refundId, "FAILED");
        // 触发人工介入告警
    }
}
```

###### 8. 如何设计一个推荐系统？

推荐系统的核心：**根据用户的历史行为，预测用户可能感兴趣的内容/商品**。

**常见推荐算法：**

**① 协同过滤（User-CF / Item-CF）**

- User-CF：找与当前用户行为相似的用户，推荐他们喜欢的内容
- Item-CF：根据当前用户喜欢的物品，找与之相似的物品（"买了XX的用户还买了YY"）

适合：电商购物推荐、电影推荐（Netflix 早期方案）。缺点：冷启动问题（新用户无历史数据）。

**② 内容推荐（Content-Based）**

分析用户历史互动的内容特征（类目、标签、关键词），推荐特征相似的内容：

```python
# 用户画像 = 历史浏览商品的类目分布
user_profile = {"手机": 0.6, "耳机": 0.3, "数码配件": 0.1}

# 找特征与用户画像匹配度高的商品
similarity = cosine_similarity(user_profile, product_feature_vector)
```

**③ 向量化召回 + 精排（现代推荐的标准架构）**

```
用户行为 → 实时特征 → Embedding 向量 → ANN 召回（相关候选集 Top 1000）
                                              ↓
                                        粗排（简单模型快速过滤，保留200）
                                              ↓
                                        精排（复杂模型打分，保留50）
                                              ↓
                                        重排（业务规则：广告插入/多样性）
                                              ↓
                                        展示给用户
```

**④ Java 后端推荐服务（在线部分）**

```java
// 推荐接口：给用户返回推荐商品列表
public List<Product> recommend(Long userId, int size) {
    // 1. 召回：从多路召回候选集中获取（协同过滤 + 热门商品 + 兴趣标签）
    List<Long> candidates = multiSourceRecall(userId, size * 10);
    
    // 2. 过滤：去除已购买/已展示过的商品
    candidates = filterSeen(userId, candidates);
    
    // 3. 排序：机器学习模型打分（调用 Python 推理服务）
    List<ScoredProduct> scored = rankingService.score(userId, candidates);
    
    // 4. 取 Top N
    return scored.stream()
        .sorted(Comparator.comparingDouble(ScoredProduct::getScore).reversed())
        .limit(size)
        .map(sp -> productService.getById(sp.getProductId()))
        .collect(toList());
}
```

###### 9. 如何设计一个风控系统？

风控系统的目标：**识别并阻止欺诈行为（薅羊毛、刷单、账号盗用）、保护业务和用户资产安全**。

**风控层次：**

```
事前：身份认证（注册/登录风险识别）
事中：交易拦截（实时规则引擎，毫秒级决策）
事后：稽查追回（离线分析，发现漏网之鱼）
```

**风控决策引擎（规则 + 模型）：**

```java
// 规则引擎（Drools / 自研）：快速判断明确的欺诈特征
public RiskDecision evaluate(TransactionContext ctx) {
    // 规则1：IP 在黑名单中
    if (blacklistService.isBlackIp(ctx.getIp())) {
        return RiskDecision.reject("IP黑名单");
    }
    
    // 规则2：1小时内失败登录超过5次
    if (loginFailCounter.count(ctx.getUserId(), 1, TimeUnit.HOURS) > 5) {
        return RiskDecision.reject("频繁登录失败");
    }
    
    // 规则3：单日累计消费超过10万
    if (consumptionService.dailyTotal(ctx.getUserId()).compareTo(100000) > 0) {
        return RiskDecision.verify("大额交易需要二次验证");
    }
    
    // 规则4：机器学习模型打分
    double fraudScore = mlModel.predict(ctx.toFeatureVector());
    if (fraudScore > 0.9) {
        return RiskDecision.reject("模型判定欺诈风险");
    }
    
    return RiskDecision.pass();
}
```

**风控决策结果：**
- `PASS`：直接通过
- `VERIFY`：需要二次验证（短信验证码、人脸识别）
- `REJECT`：直接拒绝
- `REVIEW`：人工审核

###### 10. 如何设计一个对账系统？

对账是金融系统的核心，用于发现资金和数据的不一致，保证财务准确性。

**对账类型：**

1. **内外部对账**：我方系统记录 vs 第三方（支付宝/微信支付）的流水
2. **内部对账**：订单系统 vs 支付系统的数据一致性
3. **实时对账**：每笔交易实时比对（成本高）
4. **批量对账**：T+1（次日）批量对账（主流方案）

**T+1 批量对账流程：**

```java
// 每天凌晨2点执行对账
@Scheduled(cron = "0 0 2 * * ?")
public void reconcile() {
    LocalDate reconcileDate = LocalDate.now().minusDays(1); // 对前一天
    
    // 1. 获取内部支付流水
    List<Payment> internalPayments = paymentMapper.findByDate(reconcileDate);
    
    // 2. 获取第三方流水（从支付宝/微信下载对账文件）
    List<ThirdPartyTrade> externalTrades = alipayClient.downloadBill(reconcileDate);
    
    // 3. 比对
    Map<String, Payment> internalMap = internalPayments.stream()
        .collect(toMap(p -> p.getPayNo(), p -> p));
    
    for (ThirdPartyTrade trade : externalTrades) {
        Payment internal = internalMap.get(trade.getOutTradeNo());
        
        if (internal == null) {
            // 内部有记录，第三方无 → 单边账（可能是第三方未回调）
            recordDiscrepancy(trade, DiscrepancyType.EXTERNAL_ONLY);
        } else if (!internal.getAmount().equals(trade.getTotalAmount())) {
            // 金额不一致
            recordDiscrepancy(trade, internal, DiscrepancyType.AMOUNT_MISMATCH);
        } else {
            // 一致，标记已对账
            paymentMapper.markReconciled(internal.getId());
        }
    }
    
    // 4. 内部有但第三方没有的（可能是内部异常）
    Set<String> externalTradeNos = externalTrades.stream()
        .map(t -> t.getOutTradeNo()).collect(toSet());
    internalPayments.stream()
        .filter(p -> !externalTradeNos.contains(p.getPayNo()))
        .forEach(p -> recordDiscrepancy(p, DiscrepancyType.INTERNAL_ONLY));
}
```

**差异处理：**

- 单边账（内部多/少）：人工审核，根据实际情况补录或冲销
- 金额不一致：第一时间告警，涉及资损立即升级
- 自动补偿：确认是系统 Bug 导致的，自动补发成功/失败通知

**对账报告：** 每天生成对账报告：总笔数、总金额、差异笔数/金额，发邮件给财务部门。
