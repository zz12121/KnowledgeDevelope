###### 1. MongoDB 的常用 CRUD 操作有哪些？

```javascript
// ===== 插入 =====
db.users.insertOne({ name: "张三", age: 25 })
db.users.insertMany([{ name: "李四" }, { name: "王五" }])

// ===== 查询 =====
db.users.find()                                    // 查全部
db.users.find({ age: { $gte: 18 } })               // 条件查询
db.users.find({ status: "active" }, { name: 1, _id: 0 })  // 投影（只返回 name）
db.users.findOne({ name: "张三" })                 // 查一条
db.users.find().sort({ age: -1 }).skip(0).limit(10) // 分页

// ===== 更新 =====
db.users.updateOne({ name: "张三" }, { $set: { age: 26 } })  // 更新一条
db.users.updateMany({ status: "inactive" }, { $set: { archived: true } })
db.users.replaceOne({ _id: id }, { name: "新文档" })          // 全量替换
db.users.findOneAndUpdate({ name: "张三" }, { $inc: { loginCount: 1 } }, { returnDocument: "after" })

// ===== 删除 =====
db.users.deleteOne({ _id: id })
db.users.deleteMany({ status: "deleted" })
db.users.findOneAndDelete({ _id: id })  // 删并返回被删文档

// ===== 统计 =====
db.users.countDocuments({ status: "active" })
db.users.estimatedDocumentCount()  // 快速估算总数（不保证精确）
```

###### 2. 如何在 MongoDB 中进行批量操作？

```javascript
// bulkWrite：批量混合操作，比逐条操作性能好得多
db.users.bulkWrite([
    { insertOne: { document: { name: "批量用户1" } } },
    { updateOne: {
        filter: { name: "张三" },
        update: { $set: { age: 30 } }
    }},
    { deleteOne: { filter: { status: "deleted" } } },
    { replaceOne: {
        filter: { _id: ObjectId("...") },
        replacement: { name: "新文档" }
    }}
], { ordered: false })  // ordered: false 允许并行执行（某条失败不影响其他）
```

**性能对比**：1000 条 `insertOne` vs 1 次 `insertMany`：单次批量操作减少了网络往返和服务器解析开销，性能通常提升 10-100 倍。

###### 3. MongoDB 的聚合框架（Aggregation）是什么？

聚合管道（Aggregation Pipeline）是 MongoDB 的数据处理框架，将文档按顺序通过多个**阶段（Stage）**进行变换、过滤、计算：

```javascript
db.orders.aggregate([
    // 阶段1：过滤（先过滤减少数据量，最重要！）
    { $match: { status: "completed", createdAt: { $gte: ISODate("2024-01-01") } } },
    
    // 阶段2：按用户分组，计算总金额
    { $group: { _id: "$userId", totalAmount: { $sum: "$amount" }, orderCount: { $sum: 1 } } },
    
    // 阶段3：过滤（总消费 > 1000）
    { $match: { totalAmount: { $gte: 1000 } } },
    
    // 阶段4：关联用户信息
    { $lookup: {
        from: "users",
        localField: "_id",
        foreignField: "_id",
        as: "userInfo"
    }},
    
    // 阶段5：展开关联的用户数组
    { $unwind: "$userInfo" },
    
    // 阶段6：只保留需要的字段
    { $project: { userName: "$userInfo.name", totalAmount: 1, orderCount: 1 } },
    
    // 阶段7：按总金额降序
    { $sort: { totalAmount: -1 } },
    
    // 阶段8：取前 10 名
    { $limit: 10 }
])
```

---

###### 4. 聚合管道中常用的操作符有哪些？

| 阶段操作符 | 作用 |
|-----------|------|
| `$match` | 过滤文档（类似 WHERE），**越早用越好** |
| `$group` | 按字段分组，进行聚合计算（SUM/COUNT/AVG/MAX/MIN） |
| `$project` | 字段投影/重命名/新增计算字段 |
| `$sort` | 排序 |
| `$limit` | 限制数量 |
| `$skip` | 跳过数量（分页） |
| `$lookup` | 关联其他集合（类似 LEFT JOIN） |
| `$unwind` | 展开数组，每个元素变成一个独立文档 |
| `$addFields` | 向文档添加新字段（不改变原有字段） |
| `$bucket` | 按范围分桶（如年龄 0-18、18-35、35+） |
| `$facet` | 多维度并行聚合（一次查询同时计算多个统计） |
| `$out` | 将聚合结果写入新集合（数据处理 ETL） |
| `$merge` | 将聚合结果合并到已有集合（4.2+） |

###### 5. 如何在 MongoDB 中进行分页查询？

```javascript
// 方案一：skip + limit（简单，但大页码性能差）
db.articles.find().sort({ _id: 1 }).skip((page-1)*20).limit(20)
// 缺点：skip 100000 仍要扫描前 100000 条！

// 方案二：基于游标分页（推荐，大数据集性能稳定）
// 第一页
const firstPage = db.articles.find().sort({ _id: 1 }).limit(20).toArray();
const lastId = firstPage[firstPage.length-1]._id;

// 下一页（用最后一条的 _id 作为起始点）
const nextPage = db.articles.find({ _id: { $gt: lastId } }).sort({ _id: 1 }).limit(20).toArray();
// 性能：O(log n)，无论在第几页都一样快
```

###### 6. MongoDB 的 MapReduce 是什么？

MapReduce 是 MongoDB 的数据处理方式之一（基于 JavaScript），但已被**聚合管道替代**（聚合管道更快、更易优化）。

```javascript
// MapReduce 示例（不推荐，仅了解）
db.orders.mapReduce(
    function() { emit(this.userId, this.amount); },  // Map：发出 key-value
    function(key, values) { return Array.sum(values); },  // Reduce：合并
    { out: "orderTotals" }
)
// 推荐用聚合管道替代：
db.orders.aggregate([{ $group: { _id: "$userId", total: { $sum: "$amount" } } }])
```

###### 7. 如何在 MongoDB 中进行全文搜索？

```javascript
// 1. 创建文本索引
db.articles.createIndex({ title: "text", content: "text" })
// 权重（title 比 content 更重要）
db.articles.createIndex({ title: "text", content: "text" }, { weights: { title: 10, content: 1 } })

// 2. 全文搜索（$text + $search）
db.articles.find({ $text: { $search: "MongoDB 索引" } })
// 多个词是 OR 关系；加引号是精确匹配；- 号排除词
db.articles.find({ $text: { $search: "\"MongoDB\" 索引 -PostgreSQL" } })

// 3. 按相关度排序
db.articles.find(
    { $text: { $search: "MongoDB" } },
    { score: { $meta: "textScore" } }  // 投影显示相关度分数
).sort({ score: { $meta: "textScore" } })
```

**注意**：MongoDB 全文搜索功能有限（不支持中文分词、近义词等）。生产环境推荐用 Elasticsearch 做全文搜索，MongoDB 做主存储。

###### 8. MongoDB 的 $lookup 操作符是什么？

`$lookup` 是聚合管道中实现集合间关联（类似 SQL LEFT JOIN）的操作符：

```javascript
// 基础 $lookup
db.orders.aggregate([
    { $lookup: {
        from: "users",           // 关联集合
        localField: "userId",    // 本集合字段
        foreignField: "_id",     // 关联集合字段
        as: "user"               // 结果字段名（数组）
    }},
    { $unwind: "$user" }         // 展开（一对一关系时）
])

// 管道式 $lookup（3.6+，支持复杂条件）
db.orders.aggregate([
    { $lookup: {
        from: "products",
        let: { orderItems: "$items" },  // 定义变量
        pipeline: [
            { $match: { $expr: { $in: ["$_id", "$$orderItems"] } } },
            { $project: { name: 1, price: 1 } }
        ],
        as: "productDetails"
    }}
])
```

**$lookup 性能注意**：
- 确保关联字段有索引（`foreignField` 上的索引）
- 避免大集合 $lookup 大集合，结果集可能超内存限制
- 能用嵌入文档解决的，优先用嵌入，避免 $lookup

###### 追问：$group 中有哪些常用聚合累加器？

```javascript
$sum         // 求和：{ $sum: "$amount" } 或 { $sum: 1 }（计数）
$avg         // 平均值
$max / $min  // 最大/最小值
$first / $last   // 第一条/最后一条（按文档顺序）
$push        // 将所有值推入数组
$addToSet    // 将值推入数组（去重）
$count       // 计数（5.0+）：{ $count: {} }
$stdDevPop   // 总体标准差
$stdDevSamp  // 样本标准差
```
