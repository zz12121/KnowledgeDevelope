# 策略模式（Strategy）

> 定义一组算法，把每个算法封装成独立类，使它们可以相互替换。策略模式让算法的变化独立于使用算法的客户端。核心：Context 持有 Strategy 接口的引用，运行时可以动态切换。

## 一、定义与意图

定义一组算法，把每个算法封装成独立类，使它们可以相互替换。策略模式让算法的变化独立于使用算法的客户端。核心：Context 持有 Strategy 接口的引用，运行时可以动态切换。

## 二、Demo

```java
// 场景：电商平台多种促销策略
public interface PromotionStrategy {
    BigDecimal calculate(BigDecimal originalPrice, Order order);
    String getName();
    String getCode();
}

// 满减策略
@Component("fullReduceStrategy")
public class FullReduceStrategy implements PromotionStrategy {
    private final BigDecimal threshold;
    private final BigDecimal reduceAmount;

    public FullReduceStrategy() {
        this.threshold = new BigDecimal("200");
        this.reduceAmount = new BigDecimal("30");
    }

    @Override
    public BigDecimal calculate(BigDecimal originalPrice, Order order) {
        if (originalPrice.compareTo(threshold) >= 0) {
            return originalPrice.subtract(reduceAmount);
        }
        return originalPrice;
    }

    @Override
    public String getName() { return "满200减30"; }

    @Override
    public String getCode() { return "FULL_REDUCE"; }
}

// 折扣策略
@Component("discountStrategy")
public class DiscountStrategy implements PromotionStrategy {
    @Override
    public BigDecimal calculate(BigDecimal originalPrice, Order order) {
        // 会员九折，非会员九五折
        BigDecimal rate = order.isMember() ? new BigDecimal("0.9") : new BigDecimal("0.95");
        return originalPrice.multiply(rate).setScale(2, RoundingMode.HALF_UP);
    }

    @Override
    public String getName() { return "会员折扣"; }

    @Override
    public String getCode() { return "MEMBER_DISCOUNT"; }
}

// 新人优惠券
@Component("newUserCouponStrategy")
public class NewUserCouponStrategy implements PromotionStrategy {
    @Override
    public BigDecimal calculate(BigDecimal originalPrice, Order order) {
        if (order.isNewUser()) {
            return originalPrice.subtract(new BigDecimal("50")).max(BigDecimal.ZERO);
        }
        return originalPrice;
    }

    @Override
    public String getName() { return "新人专享-50元"; }

    @Override
    public String getCode() { return "NEW_USER_COUPON"; }
}

// 策略上下文：Spring 自动注入所有策略
@Component
public class PromotionContext {
    private final Map<String, PromotionStrategy> strategies;

    @Autowired
    public PromotionContext(List<PromotionStrategy> strategyList) {
        this.strategies = strategyList.stream()
                .collect(Collectors.toMap(PromotionStrategy::getCode, s -> s));
    }

    public BigDecimal applyPromotion(String promotionCode, BigDecimal price, Order order) {
        PromotionStrategy strategy = strategies.getOrDefault(promotionCode,
                (p, o) -> p); // 默认策略：不优惠
        BigDecimal finalPrice = strategy.calculate(price, order);
        System.out.printf("促销[%s]：¥%.2f -> ¥%.2f%n", promotionCode, price, finalPrice);
        return finalPrice;
    }

    public List<String> getAvailablePromotions(Order order) {
        return strategies.values().stream()
                .map(PromotionStrategy::getName)
                .collect(Collectors.toList());
    }
}
```
