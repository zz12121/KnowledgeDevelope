# 状态模式（State）

> 允许对象在**内部状态改变时改变行为**，把每种状态下的行为封装到独立的状态类，消除大量 if-else 状态判断。

---

## 故事引入

### 《小王的糖果售卖机》

小王发现商场儿童乐园周围有很多小朋友喜欢吃糖果，于是设计了一台糖果售卖机。这台机器有以下几个状态：
- **无币状态**：等待投币
- **已投币状态**：已投币，等待选择糖果
- **出货中状态**：正在出货
- **售罄状态**：糖果卖完了

不同状态下，同样的"投币"动作会有不同结果。

**状态模式**：允许对象在内部状态改变时改变它的行为。

---

## 一、定义与意图

允许对象在**内部状态改变时改变行为**，把每种状态下的行为封装到独立的状态类，消除大量 if-else 状态判断。

## 二、Demo

```java
// 场景：自动贩卖机状态机
public interface VendingMachineState {
    void insertCoin(VendingMachine machine, int amount);
    void selectProduct(VendingMachine machine, String product);
    void dispense(VendingMachine machine);
    void cancel(VendingMachine machine);
    String getStateName();
}

// 空闲状态（等待投币）
public class IdleState implements VendingMachineState {
    @Override
    public void insertCoin(VendingMachine machine, int amount) {
        machine.addBalance(amount);
        System.out.println("已投入 " + amount + " 元，当前余额: " + machine.getBalance() + " 元");
        machine.setState(new CoinInsertedState());
    }
    @Override
    public void selectProduct(VendingMachine m, String p) { System.out.println("请先投币"); }
    @Override
    public void dispense(VendingMachine m) { System.out.println("请先投币并选择商品"); }
    @Override
    public void cancel(VendingMachine m) { System.out.println("没有可退款项"); }
    @Override
    public String getStateName() { return "空闲"; }
}

// 已投币状态
public class CoinInsertedState implements VendingMachineState {
    @Override
    public void insertCoin(VendingMachine machine, int amount) {
        machine.addBalance(amount);
        System.out.println("继续投入 " + amount + " 元，当前余额: " + machine.getBalance() + " 元");
    }
    @Override
    public void selectProduct(VendingMachine machine, String product) {
        int price = machine.getPrice(product);
        if (machine.getBalance() < price) {
            System.out.println("余额不足，商品价格 " + price + " 元，当前余额 " + machine.getBalance() + " 元");
            return;
        }
        if (!machine.hasStock(product)) {
            System.out.println("商品 " + product + " 已售罄");
            return;
        }
        machine.setSelectedProduct(product);
        machine.setState(new ProductSelectedState());
        System.out.println("已选择：" + product + "（¥" + price + "）");
    }
    @Override
    public void dispense(VendingMachine m) { System.out.println("请先选择商品"); }
    @Override
    public void cancel(VendingMachine machine) {
        System.out.println("已退款: " + machine.getBalance() + " 元");
        machine.clearBalance();
        machine.setState(new IdleState());
    }
    @Override
    public String getStateName() { return "已投币"; }
}

// 已选商品状态
public class ProductSelectedState implements VendingMachineState {
    @Override
    public void insertCoin(VendingMachine m, int a) { System.out.println("已选商品，请出货或取消"); }
    @Override
    public void selectProduct(VendingMachine m, String p) { System.out.println("已选商品，请等待出货"); }
    @Override
    public void dispense(VendingMachine machine) {
        String product = machine.getSelectedProduct();
        int price = machine.getPrice(product);
        machine.deductBalance(price);
        machine.reduceStock(product);
        System.out.printf("出货：%s，找零: %d 元%n", product, machine.getBalance());
        machine.clearBalance();
        machine.setState(new IdleState());
    }
    @Override
    public void cancel(VendingMachine machine) {
        System.out.println("已取消，退款: " + machine.getBalance() + " 元");
        machine.clearBalance();
        machine.setState(new IdleState());
    }
    @Override
    public String getStateName() { return "已选商品"; }
}

// 上下文：自动贩卖机
public class VendingMachine {
    private VendingMachineState currentState = new IdleState();
    private int balance = 0;
    private String selectedProduct;
    private final Map<String, Integer> products = Map.of("可乐", 3, "水", 2, "咖啡", 5);
    private final Map<String, Integer> stock = new HashMap<>(Map.of("可乐", 10, "水", 20, "咖啡", 5));

    public void setState(VendingMachineState state) {
        System.out.println("状态: " + currentState.getStateName() + " -> " + state.getStateName());
        this.currentState = state;
    }

    // 委托给当前状态
    public void insertCoin(int amount) { currentState.insertCoin(this, amount); }
    public void selectProduct(String product) { currentState.selectProduct(this, product); }
    public void dispense() { currentState.dispense(this); }
    public void cancel() { currentState.cancel(this); }

    // 数据方法
    public int getBalance() { return balance; }
    public void addBalance(int a) { balance += a; }
    public void deductBalance(int a) { balance -= a; }
    public void clearBalance() { balance = 0; }
    public int getPrice(String p) { return products.getOrDefault(p, Integer.MAX_VALUE); }
    public boolean hasStock(String p) { return stock.getOrDefault(p, 0) > 0; }
    public void reduceStock(String p) { stock.merge(p, -1, Integer::sum); }
    public String getSelectedProduct() { return selectedProduct; }
    public void setSelectedProduct(String p) { this.selectedProduct = p; }
}
```
