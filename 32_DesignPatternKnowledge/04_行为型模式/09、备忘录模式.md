## 二、Demo

```java
// 场景：代码编辑器的撤销/重做功能
public class CodeEditor {
    private String content;
    private int cursorPosition;

    public CodeEditor() { this.content = ""; this.cursorPosition = 0; }

    // 原发器：创建备忘录（快照）
    public EditorSnapshot createSnapshot() {
        return new EditorSnapshot(content, cursorPosition, LocalDateTime.now());
    }

    // 原发器：从备忘录恢复
    public void restore(EditorSnapshot snapshot) {
        this.content = snapshot.getContent();
        this.cursorPosition = snapshot.getCursorPosition();
        System.out.println("已恢复到: " + snapshot.getTimestamp());
    }

    public void type(String text) {
        content = content.substring(0, cursorPosition) + text + content.substring(cursorPosition);
        cursorPosition += text.length();
    }

    public void deleteChars(int count) {
        int start = Math.max(0, cursorPosition - count);
        content = content.substring(0, start) + content.substring(cursorPosition);
        cursorPosition = start;
    }

    public String getContent() { return content; }
}

// 备忘录：存储编辑器状态（只允许 CodeEditor 创建，外部只能持有引用）
public final class EditorSnapshot {
    private final String content;
    private final int cursorPosition;
    private final LocalDateTime timestamp;

    EditorSnapshot(String content, int cursorPosition, LocalDateTime timestamp) {
        this.content = content; this.cursorPosition = cursorPosition;
        this.timestamp = timestamp;
    }

    String getContent() { return content; }            // 包级别访问，外部无法直接读
    int getCursorPosition() { return cursorPosition; }
    public LocalDateTime getTimestamp() { return timestamp; }
}

// 负责人：历史记录管理（不关心备忘录内容，只管存取）
public class EditorHistory {
    private final Deque<EditorSnapshot> undoStack = new ArrayDeque<>();
    private final Deque<EditorSnapshot> redoStack = new ArrayDeque<>();
    private final int maxHistory;

    public EditorHistory(int maxHistory) { this.maxHistory = maxHistory; }

    public void save(EditorSnapshot snapshot) {
        undoStack.push(snapshot);
        redoStack.clear(); // 新操作后清空 redo 栈
        if (undoStack.size() > maxHistory) undoStack.pollLast();
    }

    public Optional<EditorSnapshot> undo() {
        if (undoStack.isEmpty()) return Optional.empty();
        EditorSnapshot current = undoStack.pop();
        redoStack.push(current);
        return Optional.ofNullable(undoStack.peek()); // 返回上一个状态
    }

    public Optional<EditorSnapshot> redo() {
        if (redoStack.isEmpty()) return Optional.empty();
        EditorSnapshot snapshot = redoStack.pop();
        undoStack.push(snapshot);
        return Optional.of(snapshot);
    }
}
```
