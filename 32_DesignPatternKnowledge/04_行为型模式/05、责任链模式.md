# 责任链模式（Chain of Responsibility）

> 将请求的处理者组成一条链，每个处理者决定自己是否处理该请求，不处理则沿链传递。客户端只需把请求发给链的头部，不需要知道哪个处理者会最终响应。

## 故事引入

### 《小王看病记》

小王生病了，去医院看病：
1. 先去分诊台（护士判断该挂什么科）
2.  去相应科室（医生诊断）
3. 需要验血（检验科抽血化验）
4. 需要拍片（放射科拍CT）
5. ……

每个环节的医生/护士只处理自己职责范围内的事，处理不了的交给下一个环节。

**责任链模式**：把请求从链中的一个对象传到下一个对象，直到请求被处理。

## 一、定义与意图

将请求的处理者组成一条链，每个处理者决定自己是否处理该请求，不处理则沿链传递。客户端只需把请求发给链的头部，不需要知道哪个处理者会最终响应。

## 二、Demo

```java
// 场景：API 请求处理管道（认证→鉴权→限流→日志→业务）
public interface Filter {
    boolean doFilter(ApiRequest request, ApiResponse response, FilterChain chain);
    int getOrder();
}

public class FilterChain {
    private final List<Filter> filters;
    private int currentIndex = 0;

    public FilterChain(List<Filter> filters) {
        this.filters = filters.stream()
                .sorted(Comparator.comparingInt(Filter::getOrder))
                .collect(Collectors.toList());
    }

    public boolean proceed(ApiRequest request, ApiResponse response) {
        if (currentIndex < filters.size()) {
            Filter filter = filters.get(currentIndex++);
            return filter.doFilter(request, response, this);
        }
        return true; // 所有过滤器都通过
    }
}

// 过滤器1：JWT 认证（最先执行）
@Component
public class JwtAuthFilter implements Filter {
    @Override
    public boolean doFilter(ApiRequest request, ApiResponse response, FilterChain chain) {
        String token = request.getHeader("Authorization");
        if (token == null || !token.startsWith("Bearer ")) {
            response.setStatus(401);
            response.setBody("{\"error\":\"Missing or invalid token\"}");
            return false; // 中断链
        }
        try {
            Claims claims = JwtUtils.parse(token.substring(7));
            request.setAttribute("userId", claims.getSubject());
            return chain.proceed(request, response); // 继续传递
        } catch (Exception e) {
            response.setStatus(401);
            return false;
        }
    }

    @Override
    public int getOrder() { return 1; }
}

// 过滤器2：权限检查
@Component
public class AuthorizationFilter implements Filter {
    @Override
    public boolean doFilter(ApiRequest request, ApiResponse response, FilterChain chain) {
        String userId = (String) request.getAttribute("userId");
        String requiredPermission = getRequiredPermission(request.getPath());
        if (!hasPermission(userId, requiredPermission)) {
            response.setStatus(403);
            response.setBody("{\"error\":\"Forbidden\"}");
            return false;
        }
        return chain.proceed(request, response);
    }

    private String getRequiredPermission(String path) { return "USER_ACCESS"; }
    private boolean hasPermission(String userId, String permission) { return true; }

    @Override
    public int getOrder() { return 2; }
}

// 过滤器3：限流
@Component
public class RateLimitFilter implements Filter {
    private final Map<String, AtomicInteger> requestCount = new ConcurrentHashMap<>();

    @Override
    public boolean doFilter(ApiRequest request, ApiResponse response, FilterChain chain) {
        String ip = request.getClientIp();
        int count = requestCount.computeIfAbsent(ip, k -> new AtomicInteger(0)).incrementAndGet();
        if (count > 100) { // 每分钟100次
            response.setStatus(429);
            response.setBody("{\"error\":\"Too Many Requests\"}");
            return false;
        }
        return chain.proceed(request, response);
    }

    @Override
    public int getOrder() { return 3; }
}

// 过滤器4：请求日志（最后执行）
@Component
public class RequestLoggingFilter implements Filter {
    @Override
    public boolean doFilter(ApiRequest request, ApiResponse response, FilterChain chain) {
        long start = System.currentTimeMillis();
        boolean result = chain.proceed(request, response);
        System.out.printf("[%s] %s -> %d (%dms)%n",
                request.getMethod(), request.getPath(), response.getStatus(),
                System.currentTimeMillis() - start);
        return result;
    }

    @Override
    public int getOrder() { return Integer.MAX_VALUE; } // 日志最后执行
}
```
