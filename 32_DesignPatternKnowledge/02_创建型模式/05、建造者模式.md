# 建造者模式（Builder）

> 将复杂对象的**构建过程**与其**表示分离**，使得同样的构建过程可以创建不同的表示。适合：参数多（特别是可选参数多）、需要分步骤构建、需要不可变对象。

## 故事引入

### 《网红小王买车记》

网红小王想买一辆车，他对销售说："我要发动机强劲的、车身要骚红色的、内饰要真皮的、轮毂要21寸的、音响要哈曼卡顿的……"

销售把小王的要求一一记录，然后让工厂按照这些要求组装——这就是**建造者模式**：将一个复杂对象的构建过程和它的表示分离。

## 一、定义与意图

将复杂对象的**构建过程**与其**表示分离**，使得同样的构建过程可以创建不同的表示。适合：参数多（特别是可选参数多）、需要分步骤构建、需要不可变对象。

## 二、类图角色

```

├── + buildPartB()

ConcreteBuilder implements Builder
```

```java
// 电商平台的查询条件构建器
public class ProductQueryCondition {
    // 所有字段都是 final，保证对象不可变
    private final String keyword;
    private final Long categoryId;
    private final BigDecimal minPrice;
    private final BigDecimal maxPrice;
    private final Integer minStock;
    private final String sortField;
    private final String sortOrder;
    private final int pageNum;
    private final int pageSize;
    private final boolean includeOffShelf;

    private ProductQueryCondition(Builder builder) {
        this.keyword = builder.keyword;
        this.categoryId = builder.categoryId;
        this.minPrice = builder.minPrice;
        this.maxPrice = builder.maxPrice;
        this.minStock = builder.minStock;
        this.sortField = builder.sortField;
        this.sortOrder = builder.sortOrder;
        this.pageNum = builder.pageNum;
        this.pageSize = builder.pageSize;
        this.includeOffShelf = builder.includeOffShelf;
    }

    // getters...
    public String getKeyword() { return keyword; }
    public Long getCategoryId() { return categoryId; }
    public BigDecimal getMinPrice() { return minPrice; }
    public BigDecimal getMaxPrice() { return maxPrice; }

    public static class Builder {
        // 必填参数
        private final int pageNum;
        private final int pageSize;
        // 可选参数，有默认值
        private String keyword;
        private Long categoryId;
        private BigDecimal minPrice;
        private BigDecimal maxPrice;
        private Integer minStock;
        private String sortField = "created_time";
        private String sortOrder = "DESC";
        private boolean includeOffShelf = false;

        public Builder(int pageNum, int pageSize) {
            if (pageNum < 1) throw new IllegalArgumentException("页码不能小于1");
            if (pageSize < 1 || pageSize > 100) throw new IllegalArgumentException("每页数量1-100");
            this.pageNum = pageNum;
            this.pageSize = pageSize;
        }

        public Builder keyword(String keyword) { this.keyword = keyword; return this; }
        public Builder category(Long categoryId) { this.categoryId = categoryId; return this; }
        public Builder priceRange(BigDecimal min, BigDecimal max) {
            if (min != null && max != null && min.compareTo(max) > 0) {
                throw new IllegalArgumentException("最低价不能大于最高价");
            }
            this.minPrice = min; this.maxPrice = max; return this;
        }
        public Builder minStock(int minStock) { this.minStock = minStock; return this; }
        public Builder sortBy(String field, String order) {
            this.sortField = field; this.sortOrder = order; return this;
        }
        public Builder includeOffShelf() { this.includeOffShelf = true; return this; }

        public ProductQueryCondition build() {
            return new ProductQueryCondition(this);
        }
    }

    @Override
    public String toString() {
        return String.format("查询条件[关键词=%s, 分类=%s, 价格区间=[%s,%s], 页码=%d, 每页=%d]",
                keyword, categoryId, minPrice, maxPrice, pageNum, pageSize);
    }
}

// 使用：链式调用，一目了然
ProductQueryCondition condition = new ProductQueryCondition.Builder(1, 20)
        .keyword("iPhone")
        .category(100L)
        .priceRange(new BigDecimal("5000"), new BigDecimal("15000"))
        .minStock(10)
        .sortBy("sales", "DESC")
        .build();

System.out.println(condition);
```

## 四、实际业务场景

```java
// 场景：HTTP 请求构建器（类似 OkHttp 的 Request.Builder）
public class ApiRequest {
    private final String url;
    private final String method;
    private final Map<String, String> headers;
    private final Map<String, Object> params;
    private final Object body;
    private final int timeoutMs;
    private final int retries;

    private ApiRequest(Builder builder) { /* 赋值 */
        this.url = builder.url; this.method = builder.method;
        this.headers = Collections.unmodifiableMap(builder.headers);
        this.params = Collections.unmodifiableMap(builder.params);
        this.body = builder.body; this.timeoutMs = builder.timeoutMs;
        this.retries = builder.retries;
    }

    public static class Builder {
        private final String url;
        private String method = "GET";
        private final Map<String, String> headers = new LinkedHashMap<>();
        private final Map<String, Object> params = new LinkedHashMap<>();
        private Object body;
        private int timeoutMs = 5000;
        private int retries = 0;

        public Builder(String url) { this.url = url; }
        public Builder post() { this.method = "POST"; return this; }
        public Builder put() { this.method = "PUT"; return this; }
        public Builder delete() { this.method = "DELETE"; return this; }
        public Builder header(String key, String value) { headers.put(key, value); return this; }
        public Builder param(String key, Object value) { params.put(key, value); return this; }
        public Builder bearerToken(String token) { return header("Authorization", "Bearer " + token); }
        public Builder json(Object body) {
            this.body = body;
            return header("Content-Type", "application/json");
        }
        public Builder timeout(int ms) { this.timeoutMs = ms; return this; }
        public Builder retries(int n) { this.retries = n; return this; }
        public ApiRequest build() { return new ApiRequest(this); }
    }
}

// 使用
ApiRequest request = new ApiRequest.Builder("https://api.example.com/orders")
        .post()
        .bearerToken("jwt-token")
        .json(createOrderRequest)
        .timeout(10000)
        .retries(3)
        .build();
```
